def scrape_shine_data(job_title, location, num_jobs=10):
    # Step 1: Get the webpage
    base_url = "https://www.shine.com/"
    search_url = f"{base_url}job-search/{job_title}-jobs-in-{location}"
    response = requests.get(search_url)

    # Step 2: Check if the request was successful
    if response.status_code == 200:
        soup = BeautifulSoup(response.text, 'html.parser')

        # Step 3: Scrape data for the first 10 jobs
        job_data = []
        job_cards = soup.find_all('li', class_='sjsj')
        for job_card in job_cards[:num_jobs]:
            title = job_card.find('h3', class_='job-title').text.strip()
            location = job_card.find('li', class_='location').text.strip()
            company_name = job_card.find('span', class_='company-name').text.strip()
            experience_required = job_card.find('span', class_='exp-years').text.strip()

            job_data.append({
                'Job Title': title,
                'Location': location,
                'Company Name': company_name,
                'Experience Required': experience_required
            })

        # Step 4: Create a DataFrame
        df = pd.DataFrame(job_data)
        return df
    else:
        print(f"Failed to retrieve data. Status code: {response.status_code}")
        return None

        pass
    driver.close()
    return None
    
def get_page_number(driver):
    try:
        return driver.find_element_by_class_name("current").text
    except Exception as e:
        try:
            get_url = driver.current_url 
            if "page=" in get_url:
                return get_url.split("page=")[1]
        except:
            print(f"Exception in get page number: {e}")
            return 0
